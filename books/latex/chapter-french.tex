\section{Phonetic Symbols}
Use \keyword{TIPA} package to input phonetic symbols.
\lstset{language=TeX}
\begin{lstlisting}
  \usepackage{tipa}
  \usepackage{tipx}
\end{lstlisting}

\begin{lstlisting}
Consonants: \\
\textipa{b d f k l m n p s t v z g K S Z dZ tS N \textltailn} 

Semivowels: \\
\textipa{j w 4}

Oral vowels: \\
\textipa{a e i o u y \o \ \oe \ O @ E}

Nasal vowels: \\
\textipa{\~a \~o \~e}
\end{lstlisting}
