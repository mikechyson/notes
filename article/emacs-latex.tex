\begin{itemize}

    type pdflatex # or
    which pdflatex
  \end{lstlisting}
\item Add the following line into .emacs
  \begin{lstlisting}
    (setenv "PATH" (concat (getenv "PATH") ":/Library/Tex/texbin"))
  \end{lstlisting}
